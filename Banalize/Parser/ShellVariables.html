def shell_variables
  ln = code.grep(/\$\{?\w+\}?/).map(&:last).join " "

  vars = ln.scan(/\$\{?\w+\}?/)
  vars.map! { |x| x.gsub(/[${}]/,'') }
  vars.reject! { |x| x =~ /^\d$/}

  @variables = vars || []
  @variables.uniq!
end
