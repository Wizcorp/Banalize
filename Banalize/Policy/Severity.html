def self.to_i severity

  when Symbol, String
    LIST[severity.to_sym]
  when Fixnum
    severity
  end

def self.to_s
  format  = "%-20s%s"
  LIST.map do |k,v|
    sprintf format , k.to_s.humanize, v
  end.join "\n"
  
end
